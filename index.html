Salin URL setelah deploy Apps Script sebagai Web App

PENTING: Error "Failed to fetch" biasanya disebabkan oleh URL yang salah, Apps Script belum di-deploy ulang setelah edit, atau izin akses belum "Anyone". Pastikan Anda mengikuti semua langkah di bawah dengan benar.
1 Buka Google Sheet Anda

Buka sheets.google.com dan buka spreadsheet yang ingin dipakai

2 Buka Apps Script

Menu Extensions → Apps Script. Hapus semua kode default.

3 Paste kode berikut (VERSI FIX)

Copy kode di bawah, paste ke Apps Script editor:

// ===== TradeLog Apps Script v2 (FIX) ===== // Paste kode ini ke Apps Script, lalu Deploy → New deployment → Web app // Execute as: Me | Who has access: Anyone function doGet(e) { return handleRequest(e, 'GET'); } function doPost(e) { return handleRequest(e, 'POST'); } function handleRequest(e, method) { try { const action = e.parameter.action; const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName('Trades'); if (!sheet) { sheet = ss.insertSheet('Trades'); sheet.appendRow(['id','date','time','symbol','side','strategy','entry','exit','size','risk','pnl','notes','image','createdAt']); sheet.getRange(1,1,1,14).setFontWeight('bold').setBackground('#10b981').setFontColor('#ffffff'); sheet.setFrozenRows(1); } let result; if (action === 'getAll') { result = getAll(sheet); } else if (action === 'ping') { result = { success: true, message: 'OK', time: new Date().toISOString() }; } else { // POST actions - read body let bodyData = {}; if (method === 'POST' && e.postData) { try { bodyData = JSON.parse(e.postData.contents); } catch(err) {} } else if (e.parameter.data) { try { bodyData = JSON.parse(e.parameter.data); } catch(err) {} } if (action === 'create') { result = create(sheet, bodyData); } else if (action === 'update') { result = update(sheet, bodyData); } else if (action === 'delete') { result = remove(sheet, bodyData.id || e.parameter.id); } else { result = { success: false, error: 'Unknown action: ' + action }; } } return ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({ success: false, error: err.message })) .setMimeType(ContentService.MimeType.JSON); } } function getAll(sheet) { const data = sheet.getDataRange().getValues(); if (data.length <= 1) return { success: true, data: [] }; const headers = data[0]; const rows = data.slice(1).map(row => { const obj = {}; headers.forEach((h, i) => obj[h] = row[i]); ['entry','exit','size','risk','pnl'].forEach(k => { if (obj[k] !== '' && obj[k] !== undefined && obj[k] !== null) { obj[k] = Number(obj[k]); } }); return obj; }); return { success: true, data: rows }; } function create(sheet, data) { try { if (!data || !data.id) return { success: false, error: 'Missing id' }; data.createdAt = data.createdAt || new Date().toISOString(); sheet.appendRow([ data.id, data.date, data.time || '', data.symbol, data.side, data.strategy || '', data.entry, data.exit, data.size, data.risk || 0, data.pnl, data.notes || '', data.image || '', data.createdAt ]); return { success: true, id: data.id }; } catch(err) { return { success: false, error: err.message }; } } function update(sheet, data) { try { if (!data || !data.id) return { success: false, error: 'Missing id' }; const allData = sheet.getDataRange().getValues(); const headers = allData[0]; const idCol = headers.indexOf('id'); let rowIndex = -1; for (let i = 1; i < allData.length; i++) { if (String(allData[i][idCol]) === String(data.id)) { rowIndex = i + 1; break; } } if (rowIndex === -1) return { success: false, error: 'Trade not found' }; sheet.getRange(rowIndex, 1, 1, 14).setValues([[ data.id, data.date, data.time || '', data.symbol, data.side, data.strategy || '', data.entry, data.exit, data.size, data.risk || 0, data.pnl, data.notes || '', data.image || '', data.createdAt || new Date().toISOString() ]]); return { success: true }; } catch(err) { return { success: false, error: err.message }; } } function remove(sheet, id) { try { if (!id) return { success: false, error: 'Missing id' }; const allData = sheet.getDataRange().getValues(); const headers = allData[0]; const idCol = headers.indexOf('id'); let rowIndex = -1; for (let i = 1; i < allData.length; i++) { if (String(allData[i][idCol]) === String(id)) { rowIndex = i + 1; break; } } if (rowIndex === -1) return { success: false, error: 'Trade not found' }; sheet.deleteRow(rowIndex); return { success: true }; } catch(err) { return { success: false, error: err.message }; } }
4 Deploy sebagai Web App

Klik Deploy → New deployment. Pilih type Web app. Set:

  • Execute as: Me (email Anda)
  • Who has access: Anyone (WAJIB!)

Klik Deploy, authorize jika diminta, lalu salin URL Web app.

5 Test URL di browser

Buka URL di tab baru. Harus muncul {"success":true,"data":[]}. Jika error, berarti deployment gagal.

6 Paste URL di atas → Test Koneksi

Tempel URL, klik "Test Koneksi". Jika berhasil, klik "Simpan".

Troubleshooting "Failed to fetch":
  • Pastikan URL dimulai dengan https://script.google.com/macros/s/
  • Setelah edit kode di Apps Script, WAJIB klik Deploy → Manage deployments → Edit (pensil) → Version: New version → Deploy
  • Pastikan "Who has access" = Anyone (bukan "Anyone with Google account")
  • Coba buka URL langsung di browser dulu untuk verifikasi
  • Beberapa browser extension (adblocker, privacy) bisa memblokir request
